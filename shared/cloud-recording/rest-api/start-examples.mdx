```python
import requests
import json

url = "https://api.sd-rtn.com/v1/apps/:appid/cloud_recording/resourceid/:resourceid/mode/:mode/start"

headers = {
    "Authorization": "Basic <credentials>",
    "Content-Type": "application/json"
}

data = {
    "cname": "<your_channel_name>",
    "uid": "<unique_user_id>",
    "clientRequest": {
        "recordingConfig": {
            "channelType": 1,
            "streamTypes": 0,
            "maxIdleTime": 30,
            "subscribeAudioUids": ["#allstream#"],
            "subscribeUidGroup": 5
        },
        "recordingFileConfig": {
            "avFileType": ["hls", "mp4"]
        },
        "storageConfig": {
            "vendor": 2,
            "region": 3,
            "bucket": "<your_storage_bucket>",
            "accessKey": "<your_storage_access_key>",
            "secretKey": "<your_storage_secret_key>",
            "fileNamePrefix": ["quickstart"]
        }
    }
}

response = requests.post(url, headers=headers, data=json.dumps(data))

print(response.status_code)
print(response.json())
