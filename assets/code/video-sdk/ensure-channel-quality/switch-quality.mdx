        {

            Debug.Log("Switching to low-quality video");

        else
        {
            Debug.Log("No remote user in the channel");
        }

    {
        if (remoteUid > 1)

            agoraEngine.SetRemoteVideoStreamType(remoteUid, VIDEO_STREAM_TYPE.VIDEO_STREAM_HIGH);

        }
        else
        {
            Debug.Log("No remote user in the channel");
        }
