#### PHP

* **HMAC/SHA1**

    ```php
    <?php
    function assertEqual($expect, $actual)
    {
        if ($expect != $actual) {
            echo "\n assert failed";
            echo "\n expect:\n " . $expect;
            echo "\n actual:\n " . $actual;
            echo "\n";
        } else {
            echo "assert ok\n";
            echo "\n";
        }
    }
    // Get the raw request body of the message notification and calculate the signature. The requestBody in the following code is a binary byte array before deserialization and not a dictionary after deserialization.
    $request_body =
        '{"eventType":10,"noticeId":"4eb720f0-8da7-11e9-a43e53f411c2761f","notifyMs":1560408533119,"payload":{"a":"1","b":2},"productId":1}';
    $secret = "secret";
    // The value of Agora-Signature in the request header.
    $sha1 = "033c62f40f687675f17f0f41f91a40c71c0f134c";
    $res1 = hash_hmac("sha1", $request_body, $secret);
    assertEqual($res1, $sha1);
    ?>
    ```

    $secret = "secret";
    // The value of Agora-Signature-V2 in the request header.
