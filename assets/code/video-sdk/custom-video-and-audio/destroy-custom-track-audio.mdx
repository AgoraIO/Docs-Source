Override `DestroyEngine` to destroy the track before you destroy the engine:
    ```csharp
    public override void DestroyEngine()
    {
        if (agoraEngine == null)
        {
            return;
        }
        // Abort the audio that pushes audio frames in the channel.
        _pushAudioFrameThread.Abort();
        // Destroy the custom audio track.
        agoraEngine.DestroyCustomAudioTrack(audioTrackID);
        base.DestroyEngine();
    }
    ```
