import CodeBlock from '@theme/CodeBlock';

1. Create an empty folder named `test-convoai`.

    <CodeBlock language="bash" showLineNumbers>
    {`mkdir test-convoai
   cd test-convoai`}
    </CodeBlock>

2. In the `test-convoai` directory, run the following command to initialize the Go module:

    <CodeBlock language="bash" showLineNumbers>
    {`go mod init test-convoai`}
    </CodeBlock>

3. Create an empty `main.go` file in the project directory for a minimal <Vpd k="ENGINE" /> implementation.

    <CodeBlock language="bash" showLineNumbers>
    {`touch main.go`}
    </CodeBlock>

4. Run the following commands in the project directory to install the SDK and dependencies:

    <CodeBlock language="bash" showLineNumbers>
    {`# Install the Agora Go SDK
   go get -u github.com/AgoraIO-Community/agora-rest-client-go

   # Update dependencies
   go mod tidy`}
    </CodeBlock>
