    private void initializeAndJoinChannel() {
        try {
            // Create and configure RtcEngineConfig object
            RtcEngineConfig config = new RtcEngineConfig();
            config.mContext = getBaseContext();
            config.mAppId = myAppId;
            config.mEventHandler = mRtcEventHandler;
            // Create and initialize RtcEngine
            mRtcEngine = RtcEngine.create(config);
        } catch (Exception e) {
            throw new RuntimeException("Error initializing RTC engine: " + e.getMessage());
        }

        // Create and configure ChannelMediaOptions object
        ChannelMediaOptions options = new ChannelMediaOptions();
        // Set user role to BROADCASTER (host) or AUDIENCE (audience)
        options.clientRoleType = Constants.CLIENT_ROLE_BROADCASTER;
        // Set the channel profile (COMMUNICATION for voice calling, LIVE_BROADCASTING for streaming)
        options.channelProfile = Constants.CHANNEL_PROFILE_COMMUNICATION;
        // Publish audio captured by the microphone
        options.publishMicrophoneTrack = true;
        // Automatically subscribe to all audio streams
        options.autoSubscribeAudio = true;
        // Join the channel using temporary Token, channel name and UID
        // The SDK generates a random UID if you pass 0
        // The onJoinChannelSuccess callback is triggered upon success
        mRtcEngine.joinChannel(token, channelName, 0, options);

    private String[] getRequiredPermissions(){

            return new String[]{
                    Manifest.permission.RECORD_AUDIO, // Audio recording permission
                    Manifest.permission.READ_PHONE_STATE, // Read phone state permission

            };
        } else {
            return new String[]{
                    Manifest.permission.RECORD_AUDIO,
            };
        }
    }

    private boolean checkPermissions() {
        for (String permission : getRequiredPermissions()) {
            int permissionCheck = ContextCompat.checkSelfPermission(this, permission);
            if (permissionCheck != PackageManager.PERMISSION_GRANTED) {
                return false;
            }
        }
        return true;
    }

    @Override

        super.onCreate(savedInstanceState);

        // If permissions are already granted, initialize RtcEngine and join the channel
        if (checkPermissions()) {
            initializeAndJoinChannel();
        } else {
            ActivityCompat.requestPermissions(this, getRequiredPermissions(), PERMISSION_REQ_ID);
        }
    }

    // System permission request callback
    @Override
    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions, @NonNull int[] grantResults) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
