class SecondChannelDelegate: NSObject, AgoraRtcEngineDelegate {
    func rtcEngine(_ engine: AgoraRtcEngineKit, didJoinChannel channel: String, withUid uid: UInt, elapsed: Int) {
        print("Join Channel Success - joined channel: \(channel) uid: \(uid)")
        secondChannelBtn.title = "Leave Second Channel"
    }

    func rtcEngine(_ engine: AgoraRtcEngineKit, didJoinedOfUid uid: UInt, elapsed: Int) {
        print("User Join Success - joined uid: \(uid)")

        let videoCanvas = AgoraRtcVideoCanvas()
        videoCanvas.uid = uid
        videoCanvas.view = remoteView
        videoCanvas.renderMode = .hidden
        engine.setupRemoteVideoEx(videoCanvas, connection: rtcSecondConnection)
    }
}
