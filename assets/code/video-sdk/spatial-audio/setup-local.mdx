        let pos: [NSNumber]     = [0, 0, 0]
        let right: [NSNumber]   = [1, 0, 0]
        let up: [NSNumber]      = [0, 1, 0]
        let forward: [NSNumber] = [0, 0, -1]

            axisForward: forward,
            axisRight: right,
            axisUp: up
        )
