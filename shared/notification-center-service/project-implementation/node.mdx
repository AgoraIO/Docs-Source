#### Node.js

* **HMAC/SHA1**
    ```java
    const crypto = require('crypto')
    // Get the raw request body of the message notification and calculate the signature.The requestBody in the following code is a binary byte array before deserialization and not a dictionary after deserialization.
    const requestBody = '{"eventType":10,"noticeId":"4eb720f0-8da7-11e9-a43e53f411c2761f","notifyMs":1560408533119,"payload":{"a":"1","b":2},"productId ":1}'
    const secret = 'secret'
    const signature = crypto.createHmac('sha1', secret).update(requestBody,
        'utf8').digest('hex')
    console.log(signature) // 033c62f40f687675f17f0f41f91a40c71c0f134c
    ```
