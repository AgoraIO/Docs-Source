To modify raw audio frames before publishing, follow these steps:

1. Create a local audio track publish it in the channel.
1. Use the Web Audio API to process the audio stream.
1. Extract audio data using an `AnalyserNode` and log it in the console.

```js
async function createAndProcessAudioTrack() {
  localAudioTrack = await AgoraRTC.createMicrophoneAudioTrack();
  await client.publish([rtc.localAudioTrack]);

  // Use Web Audio API to process the audio
  audioContext = new AudioContext();
  const source = audioContext.createMediaStreamSource(new MediaStream([rtc.localAudioTrack.getMediaStreamTrack()]));

  // Create an AnalyserNode to extract audio data
  analyser = audioContext.createAnalyser();
  analyser.fftSize = 2048; // FFT size (affects frequency resolution)
  
  // Create a buffer to store frequency data
  dataArray = new Uint8Array(analyser.frequencyBinCount);
  source.connect(analyser); // Connect the source to the analyser

  processAudioFrames(); // Start processing audio
}

function processAudioFrames() {
  if (!analyser) return;

  analyser.getByteFrequencyData(dataArray); // Get frequency data
  console.log("Audio Data:", dataArray); // This logs real-time audio frame data

  requestAnimationFrame(processAudioFrames); // Keep processing audio frames
}
```
