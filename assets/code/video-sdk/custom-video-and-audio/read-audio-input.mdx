    private fun readBuffer(): ByteArray? {
        // Read the audio file buffer
        val byteSize = bufferSize
        val buffer = ByteArray(byteSize)
        try {
            if (inputStream!!.read(buffer) < 0) {
                inputStream!!.reset()
                return readBuffer()
            }
        } catch (e: IOException) {
            e.printStackTrace()
        }
        return buffer
    }
