func updateNetworkStatus(quality: Int) {
    if (quality > 0 && quality < 3) { networkStatus.backgroundColor = NSColor.green }
    else if (quality <= 4) { networkStatus.backgroundColor = NSColor.yellow }
    else if (quality <= 6) { networkStatus.backgroundColor = NSColor.red }
    else { networkStatus.backgroundColor = NSColor.white }
}
