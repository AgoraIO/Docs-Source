const enabled = true;  // Enable or disable AINS

const result = rtcEngine.setAINSMode(enabled, AudioAinsMode.AinsModeUltralowlatency);

if (result === 0) {
    // AINS mode set successfully
} else {
    // Failed to enable AINS 
}
